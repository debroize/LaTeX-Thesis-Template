%%% ---------------------------------
% ISBN Zeilenumbruch
\newcommand{\ISBN}[1]{\breakisbn{#1}}
\newcommand{\breakisbn}[1]{%
  \begingroup
  \def\\##1{##1\hfill}%
  \def\##1{\texttt{##1}}%
  \tt\def\~{ }%
  \def\^#1{#1}%
  \hbox_to\hsize{\hfil\breakisbnaux#1\breakisbnaux}%
  \endgroup
}
\def\breakisbnaux#1 {#1\hfil\penalty0\strut}
